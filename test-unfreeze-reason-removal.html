// إزالة التحقق من الحقل const unlockReason = unlockReasonInput.value.trim(); if (!unlockReason) { alert('يرجى إدخال سبب إلغاء التجميد'); unlockReasonInput.focus(); return; } // إزالة إرسال السبب body: JSON.stringify({ unlock_reason: unlockReason }) // إزالة مسح الحقل unlockReasonInput.value = ''; // إزالة focus unlockReasonInput.focus();
